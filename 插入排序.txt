#include <iostream>
#include <math.h>
#include <cstdlib>
#include <vector>
#include <limits.h>
using namespace std;

vector<int> gen_unsorted(int array_size, int min , int max )
{
	vector<int> unSorted;
	for (int i = 0; i < array_size; ++i) {
		unSorted.push_back((rand() % (max - min + 1)) + min);  //[min,max]
	}
	return unSorted;
}

void insert_sort(vector<int> &array)
{
	int i=0,temp,key;
	int s=array.size();
    for(int j=1;j<s;j++)
	{
	    key=array[j];
		i=j-1;
	while(i>-1&&array[i]>key)
	{
		temp=array[i+1];array[i+1]=array[i];array[i]=temp;
		i=i-1;
		}
	}
	//array[s]=1000;
}

void print_vector(vector<int> vec, int begin_idx = 0, int end_idx = NULL)
{
	if (end_idx == NULL) {
		end_idx =vec.size()- 1;
	}
	for (int i = begin_idx; i <= end_idx; ++i) {
		cout << vec[i] << ", ";
	}
	cout << endl;
}


int main()
{
	int array_size=100,min=0,max=500,begin_idx=0;
	int end_idx=array_size-1;
	vector<int> array;
	array=gen_unsorted(array_size , min, max );
	print_vector( array, begin_idx , end_idx );
	cout<<endl;
	insert_sort(array);
	print_vector( array, begin_idx , end_idx );
	system("pause");
}
